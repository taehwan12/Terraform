# 1. 고가용성 서스 인스턴스 기동 템플릿 선언
resource "aws_launch_template" "asg_template" {
  name_prefix   = "4rsenal-launch-template"
  image_id      = "ami-0c9c942bd7bf113a2" # Ubuntu Server 24.04 LTS Base AMI
  instance_type = "t3.micro"

  lifecycle {
    # 인프라 업데이트 시 가용성 공백을 방지하는 선-생성 후-삭제 정책 바인딩
    create_before_destroy = true
  }

  tags = {
    Name = "terraform-launch-template"
  }
}

# 2. 동적 스케일링을 위한 Auto Scaling Group 리소스 연동
resource "aws_autoscaling_group" "pri_asg" {
  name                = "terraform-pri-asg"
  vpc_zone_identifier = [aws_subnet.pub_2a.id, aws_subnet.pub_2c.id]
  target_group_arns   = [aws_lb_target_group.alb_tg.arn]
  health_check_type   = "ELB"
  health_check_grace_period = 120 # 헬스 체크 유예 기간 설정

  # 임계치 기반 동적 가용 범위 명시
  min_size            = 2
  max_size            = 4
  desired_capacity    = 3

  launch_template {
    id      = aws_launch_template.asg_template.id
    version = "$Latest"
  }
}
